class Content

  attr_reader :data

  def initialize(incoming={})
    opts = HttpClient::Helper.symbolize_keys(incoming)
    @data = HttpClient::Preconditions.assert_class(opts.delete(:data), Array).map { |v| HttpClient::Preconditions.assert_class('data', HttpClient::Helper.to_object(v), Hash)}
  end

  def to_json
    JSON.dump(to_hash)
  end

  def to_hash
    {
      :data => data
    }
  end

end
